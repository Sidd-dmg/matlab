% Sampling Theorem Verification

% Signal parameters
f = 1000;   % Signal frequency (Hz)
fs = 4000;  % Sampling frequency (Hz)

% Time vector for continuous-time signal
t_continuous = 0:1/fs:1/f-1/fs;

% Generate continuous-time signal
x_continuous = sin(2*pi*f*t_continuous);

% Time vector for discrete-time samples
t_discrete = 0:1/fs:1/f-1/fs;

% Sample the continuous-time signal
x_discrete = sin(2*pi*f*t_discrete);

% Plot the continuous-time and discrete-time signals
subplot(2, 1, 1);
plot(t_continuous, x_continuous, 'b');
title('Continuous-Time Signal');
xlabel('Time (s)');
ylabel('Amplitude');

subplot(2, 1, 2);
stem(t_discrete, x_discrete, 'r');
title('Discrete-Time Samples');
xlabel('Time (s)');
ylabel('Amplitude');

% Verify the sampling theorem
hold on;
stem(t_discrete, x_discrete, 'r', 'filled');
plot(t_continuous, x_continuous, 'b--');
legend('Samples', 'Continuous-Time Signal');
hold off;
